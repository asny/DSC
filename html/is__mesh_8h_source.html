173  template<typename map_type, typename mesh_type>
174  inline int create_edge(int i, int j, map_type& edge_map, mesh_type& mesh)
175  {
176  int a,b;
177  if (i <= j) { a = i; b = j; }
178  else { a = j; b = i; }
179  edge_key key(a, b);
180  auto it = edge_map.find(key);
181  if (it == edge_map.end())
182  {
183  int n = mesh.insert_edge(i, j); //non-sorted
184  it = edge_map.insert(std::pair<edge_key,int>(key, n)).first;
185  }
186  return it->second;
187  }

584  template<typename value_type>
585  inline void ISMesh::for_each_par(std::function<void(value_type&,int)> fn) {
586  using KeyType = decltype(std::declval<value_type>().key()); // the type of value().key()
587  auto kernel = &get_kernel<KeyType, value_type>();
588 
589  int thread_count = std::thread::hardware_concurrency();
590 
591  std::vector<std::thread*> threads;
592  int chunk_size = (int)ceil(kernel->capacity() / (float)(thread_count));
593 
594  for (int i=0;i<thread_count;i++){
595  threads.push_back(new std::thread(run_for_each_par<KeyType, value_type>, fn, kernel, i*chunk_size,(1+i)*chunk_size, i));
596  }
597 
598  for (int i=0;i<thread_count;i++){
599  threads[i]->join();
600  delete threads[i];
601  }
602  }

604  template<typename key_type, typename value_type>
605  inline std::vector<key_type> ISMesh::find_par(std::function<bool(value_type&)> include){
606  std::vector<std::vector<key_type>> res_array(std::thread::hardware_concurrency(), {});
607  for_each_par<value_type>([&](value_type &value, int threadid){
608  if (include(value))

611  }
612  });

614  for (auto & t:res_array){
615  res.insert(res.end(), t.begin(), t.end());
616  }
617  return res;

620  template<typename value_type, typename kernel_type>
621  inline void run_for_each_par_sp(std::function<void(value_type&,int)> fn, kernel_type* kernel, double partitionsize, int max_threads, int dimension, int threadid, int actualthread){
622  for (auto &n : *kernel){
623  double p = n.get_center()[dimension];
624  double concur_partitionsize = partitionsize * max_threads;
625  if (p<0){
626  p += concur_partitionsize * (int)(ceil(-p/concur_partitionsize));
627  }
628  p = fmod(p , concur_partitionsize);
629  int run_in_thread = std::min((int)floor(p / partitionsize),max_threads-1);
630  if (run_in_thread == threadid){
631  fn(n,actualthread);
632  }
633  }
634  }

636  template<typename value_type>
637  inline void ISMesh::for_each_par_sp(double partitionsize, int dimension, std::function<void(value_type&,int)> fn) {
638  using KeyType = decltype(std::declval<value_type>().key()); // the type of value().key()
639  using KernelType = kernel<KeyType, value_type>;
640  auto kernel = &get_kernel<KeyType, value_type>();
641 
642  int thread_count = std::thread::hardware_concurrency();
643 
644  std::vector<std::thread*> threads;
645  for (int i=0;i<thread_count;i++){
646  threads.push_back(new std::thread(run_for_each_par_sp<value_type, KernelType>, fn, kernel, partitionsize,thread_count*2,dimension, i*2,i));
647  }
648  for (int i=0;i<thread_count;i++){
649  threads[i]->join();
650  }
651 
652  for (int i=0;i<thread_count;i++){
653  delete threads[i];
654  threads[i] = new std::thread(run_for_each_par_sp<value_type, KernelType>, fn, kernel, partitionsize,thread_count*2,dimension, 1+i*2,i);
655  }
656  for (int i=0;i<thread_count;i++){
657  threads[i]->join();
658  delete threads[i];
659  }
